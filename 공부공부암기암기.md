    stack = []

    # 시작점 찾기
    flag = 0

            break
        for j in range(16):
            if maze[i][j] == 2:
                sx, sy = i, j
                flag = 1
                break


    v = [sx, sy]
    stack.append(v)
    result = 0

    while (len(stack)>0):
        x, y = stack.pop()

        if result == 1:
            break

        if visited[x][y] != 1:
            visited[x][y] = 1

            for i in range(4):
                nx = x + dir[i][0]
                ny = y + dir[i][1]

                if maze[nx][ny] == 0 and visited[nx][ny] == 0:
                    stack.append([nx, ny])

                elif maze[nx][ny] == 3:
                    result = 1
                    break

    print('#{} {}'.format(testnum, result))

def notWall(x, y):
    if 0 <= x < N and 0 <= y < N:
        return True
    else:
        return False

T = int(input())
for t in range(T):
    N = int(input())
    maze = [list(map(int, input())) for _ in range(N)]
    dir = [[-1, 0], [0, 1], [0, -1], [1, 0]]
